decision_distribute_surplus = {
	owned_planets_only = yes

	enactment_time = 0
	resources = {
		categroy = decisions
		cost = {
			influence = 0
		}
	}

	potential = {
		habitable_structure = no
		NOT = {
			is_planet_class = pc_machine
		}
	}

	allow = {
		NOR = {
			has_modifier = biopopgrowth_distribute_surplus
			has_policy_flag = population_controls_allowed
			has_policy_flag = purge_allowed
			owner = {
				OR = {
					has_deficit = minerals
					has_deficit = food
				}
			}
		}
	}

	effect = {
		hidden_effect = {
			if = {
				limit = {
					NOR = {
						owner = {
							has_edict = biopopgrowth_edict_distribute_surplus
						}
						has_planet_flag = biopopgrowth_no_event_check_flag
					}
				}
				set_planet_flag = biopopgrowth_no_event_check_flag
			}
			biopopgrowth_distribute_surplus_effect = yes
		}
		add_modifier = {
			modifier = biopopgrowth_distribute_surplus
		}
	}

	ai_weight = {
		weight = 1
		modifier = {
			factor = 0
			OR = {
				free_housing < 1
				num_pops < 4
				owner = {
					OR = {
						AND = {
							has_country_resource = {
								type = minerals
								amount = 0
							}
							has_monthly_income = {
								type = minerals
								amount < 0
							}
						}
						AND = {
							has_country_resource = {
								type = food
								amount = 0
							}
							has_monthly_income = {
								type = food
								amount < 0
							}
						}
					}
				}
			}
		}
		modifier = {
			factor = 10
			owner = {
				OR = {
					AND = {
						has_country_resource = {
							type = minerals
							amount > 1000
						}
						has_country_resource = {
							type = food
							amount = 0
						}
						has_monthly_income = {
							type = food
							amount = 0
						}
					}
					has_country_resource = {
						type = food
						amount > 1000
					}
				}
			}
		}
	}
}

			is_planet_class = pc_machine
		}
	}

		has_modifier = biopopgrowth_distribute_surplus
	}

	effect = {

			biopopgrowth_remove_distribute_surplus_effect = yes

				}
				remove_planet_flag = biopopgrowth_no_event_check_flag
			}
		}
		remove_modifier = biopopgrowth_distribute_surplus
	}

	ai_weight = {

					OR = {
						AND = {
							has_country_resource = {
								type = minerals
								amount = 0
							}
							has_monthly_income = {
								type = minerals
								amount < 0
							}
						}
						AND = {
							has_country_resource = {
								type = food
								amount = 0
							}
							has_monthly_income = {
								type = food
								amount < 0
							}
						}
					}
				}
			}
		}
	}
