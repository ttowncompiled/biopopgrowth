decision_pop_census = {
	owned_planets_only = yes

	enactment_time = 28
	resources = {
		categroy = decisions
		cost = {
			influence = 0
		}
	}

	potential = {
		habitable_structure = no
		NOT = {
			is_planet_class = pc_machine
		}
	}

	effect = {
		hidden_effect = {
			biopopgrowth_count_bio_pops_effect = yes
			if = {
				limit = {
					has_modifier = biopopgrowth_pop_census
				}
				remove_modifier = biopopgrowth_pop_census
			}
			add_modifier = {
				modifier = biopopgrowth_pop_census
				days = 365
			}
		}
	}

	ai_weight = {
		weight = 1
		modifier = {
			factor = 0
			OR = {
				free_housing < 1
				num_pops < 4
				owner = {
					OR = {
						AND = {
							has_country_resource = {
								type = minerals
								amount = 0
							}
							has_monthly_income = {
								type = minerals
								amount < 0
							}
						}
						AND = {
							has_country_resource = {
								type = food
								amount = 0
							}
							has_monthly_income = {
								type = food
								amount < 0
							}
						}
					}
				}
			}
		}
		modifier = {
			factor = 10
			owner = {
				OR = {
					AND = {
						has_country_resource = {
							type = minerals
							amount > 1000
						}
						has_country_resource = {
							type = food
							amount = 0
						}
						has_monthly_income = {
							type = food
							amount = 0
						}
					}
					has_country_resource = {
						type = food
						amount > 1000
					}
				}
			}
		}
	}
}
